SELECT p.cid,cp.prename,TRIM(p.name) AS fname,TRIM(p.lname) AS lname,p.sex, DATE_FORMAT(p.BIRTH,'%Y-%m-%d') birthdate,ROUND(h.village,0) AS moo,TRIM(h.house) AS house, CONCAT(h.changwat,h.ampur,h.tambon) AS aid,p.nation AS nation,p.hospcode FROM t_person_cid p,home h,cprename cp WHERE p.hospcode=h.hospcode AND p.hid=h.hid AND p.prename=cp.id_prename AND p.typearea IN ('1','3') AND p.discharge='9' AND p.nation='099' AND h.changwat='86' GROUP BY p.hospcode,p.cid;
